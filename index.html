async function loadLeaderboard() { const listDiv = document.getElementById('list'); const timestamp = new Date().getTime(); const STEIN_URL = `https://api.steinhq.com/v1/storages/69a6fb04affba40a625861dd`; try { // 💡 캐시 방지를 위해 파라미터를 각각 붙여 호출합니다. const [statusRes, logRes] = await Promise.all([ fetch(`${STEIN_URL}/status?t=${timestamp}`), fetch(`${STEIN_URL}/log?t=${timestamp}`) ]); const statusData = await statusRes.json(); globalLogData = await logRes.json(); // 💡 10,000P 이상 오류 방지: '잔액' 데이터를 숫자로 엄격하게 변환 statusData.sort((a, b) => { const valA = parseFloat(String(a.잔액).replace(/[^0-9.-]+/g, "")) || 0; const valB = parseFloat(String(b.잔액).replace(/[^0-9.-]+/g, "")) || 0; return valB - valA; }); listDiv.innerHTML = ""; statusData.forEach((user, index) => { if(!user.이름) return; // 💡 현재 리스트에 보여줄 잔액을 명확히 정의 const currentPoints = parseFloat(String(user.잔액).replace(/[^0-9.-]+/g, "")) || 0; const rank = index + 1; const item = document.createElement('div'); item.className = 'user-item'; // 클릭 시 팝업에 현재 확정된 'currentPoints'를 그대로 넘깁니다. item.onclick = () => openModal(user.이름, currentPoints); item.innerHTML = `
${rank}
${user.이름}
${Math.floor(currentPoints).toLocaleString()} P
`; listDiv.appendChild(item); }); } catch (e) { console.error(e); listDiv.innerHTML = "
데이터 동기화 중... 잠시만 기다려주세요.
"; } }
